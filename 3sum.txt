#include <stdio.h>

void sum(int n,int *arr){
    int i,j,k;
    for( i=0;i<n-2;i++){
        for(j=i+1;j<n-1;j++){
            for(k=j+1;k<n;k++){
                if((i != j && i != k && j != k)) {
                    if((arr[i]+arr[j]+arr[k])==0){
                        printf("%d %d %d\n",i,j,k);
                    }
                }
            }
        }
    }
}

int main(){
    int arr[20],n;
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%d",&arr[i]);
    }
    sum(n,arr);
}
