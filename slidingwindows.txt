#include <stdio.h>
void sliding_windows(int *arr,int n,int k){
    int sum=0;
    for(int i=0;i<=n-k;i++){
       for(int j=i;j<i+k;j++){
           sum+=arr[j];
       } 
       printf("%d",sum);
       sum=0;
    }
}

int main() {
    int arr[20],n,k;
    scanf("%d %d",&n,&k);
    for(int i=0;i<n;i++){
        scanf("%d",&arr[i]);
    }
    sliding_windows(arr,n,k);
}
