#include <stdio.h>
void sorting (char *arr){
    int n=1,temp,j;
    while(arr[n]!='\0'){
        temp=arr[n];
        j=n-1;
        while(j>=0 && temp<arr[j]){
            arr[j+1]=arr[j];
            j--;
        }
        arr[j+1]=temp;
        n++;
    }
}
void anagram(char *arr1,char *arr2){
  int ana=1,i=0;
  sorting(arr1);
  sorting(arr2);
  while(arr1[i]!='\0'){
      if (arr1[i]!=arr2[i]) ana=0;
      i++;
  }
  if (ana==1) printf("anagram");
  else printf("not");
 
}

int main() {
  char arr1[20],arr2[20];
  scanf("%s",&arr1);
  scanf("%s",&arr2);
  anagram(arr1,arr2);
}
